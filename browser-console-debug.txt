    async function quickTest() {

            const headers = { Authorization: `Bearer ${token}` };
            
            console.log('ðŸ” Testing auth verify...');
            const authRes = await fetch('/api/auth/verify', { headers });
            const authData = await authRes.json();
            console.log('Auth verify result:', authRes.status, authData);
            
            if (authRes.status === 401) {
                console.log('âŒ AUTH FAILED - Token is invalid/expired');
                console.log('ðŸ’¡ Run this to clear token and login: localStorage.removeItem("floworx_token"); window.location.href = "/login"');
                return;
            }
            
            console.log('ðŸ” Testing user status...');
            const userRes = await fetch('/api/user/status', { headers });
            const userData = await userRes.json();
            console.log('User status result:', userRes.status, userData);
            
            console.log('ðŸ” Testing onboarding status...');
            const onboardingRes = await fetch('/api/onboarding/status', { headers });
            const onboardingData = await onboardingRes.json();
            console.log('Onboarding status result:', onboardingRes.status, onboardingData);
            
            // Check if all APIs are working
            if (authRes.ok && userRes.ok && onboardingRes.ok) {
                console.log('âœ… All APIs working! Issue might be in React component.');
                console.log('ðŸ’¡ Try refreshing the page: window.location.reload()');
            } else {
                console.log('âŒ Some APIs failed. Check the responses above.');
            }
            
        } catch (error) {
            console.log('âŒ Network error:', error);
            console.log('ðŸ’¡ Check if the backend is running');
        }
    }
    
    quickTest();
}
